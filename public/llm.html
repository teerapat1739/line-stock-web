💻 โค้ดจริง — Edge Function เรียก Claude (Anthropic SDK)

รันบน Supabase Edge Functions (Deno) — ใช้ tool use แบบบังคับ เพื่อให้ Claude คืน JSON ตาม schema เสมอ + prompt caching เพื่อลดค่าใช้จ่าย

// supabase/functions/line-webhook/llm.ts
import Anthropic from "npm:@anthropic-ai/sdk";
import { parseRecord } from "./parser.ts";

const client = new Anthropic();   // uses ANTHROPIC_API_KEY env var

// ── Schema ที่บังคับให้ Claude เติม ──────────────────────────────
const PARSE_TOOL = {
  name: "parse_stock_message",
  description: "แยกข้อความบันทึกสต็อคออกเป็น records, หรือบอกว่าเป็น chitchat / unknown",
  input_schema: {
    type: "object",
    properties: {
      kind: { type: "string", enum: ["record", "chitchat", "unknown"] },
      records: {
        type: "array",
        items: {
          type: "object",
          properties: {
            action: { type: "string", enum: ["in", "out", "adj"] },
            item:   { type: "string" },
            qty:    { type: "number" },
            unit:   { type: "string" },
            confidence: { type: "number" }     // 0-1, <0.8 → ถามยืนยัน
          },
          required: ["action", "item", "qty", "unit", "confidence"]
        }
      },
      reply: { type: "string" }       // ถ้า kind=chitchat
    },
    required: ["kind"]
  }
};

const SYSTEM = `คุณเป็น parser ของระบบบันทึกสต็อคในร้านอาหารไทย
แปลงข้อความผู้ใช้เป็น records:
  action: "in"  = เติม/เพิ่ม/มี
  action: "out" = เบิก/ออก
  action: "adj" = เกิน (ของเกินจากที่ควรมี)
ถ้าเป็นทักทาย/ขอบคุณ → kind="chitchat" + reply ภาษาไทยสุภาพ
ถ้าไม่เข้าใจ → kind="unknown"
ตัวเลขคำ ("สามร้อย"=300, "ห้าขีด"=500g, "โล/กก."=กิโลกรัม) แปลงให้

// ── Hybrid parser: regex first, LLM fallback ──────────────────────
export async function parseHybrid(text: string, knownItems: string[]) {
  // Tier 1: regex
  const quick = parseRecord(text);
  if (quick) return { source: "regex", records: [quick] };

  // Tier 2: LLM fallback
  const response = await client.messages.create({
    model: "claude-haiku-4-5",
    max_tokens: 512,
    system: [
      { type: "text", text: SYSTEM,
        cache_control: { type: "ephemeral" } },              // ① cache system prompt
      { type: "text", text: `ของในระบบ: ${knownItems.join(", ")}`,
        cache_control: { type: "ephemeral" } }               // ② cache item list
    ],
    tools: [PARSE_TOOL],
    tool_choice: { type: "tool", name: "parse_stock_message" },   // ③ บังคับ tool use
    messages: [{ role: "user", content: text }]
  });

  const block = response.content.find(b => b.type === "tool_use");
  return { source: "llm", ...(block?.input as any) };
}

หัวใจของโค้ดนี้คือ 3 จุด

  • ① + ② Prompt caching — system prompt + item list เปลี่ยนน้อย แคชไว้ครั้งละ 5 นาที ลดต้นทุน ~90%
  • ③ Forced tool use — tool_choice: {type:"tool", name:...} บังคับให้ Claude คืน JSON ตาม schema เสมอ ไม่ free-form text
  • regex ก่อน LLM ทีหลัง — ถ้า regex ผ่าน ไม่ต้องเรียก LLM เลย (ประหยัด + เร็ว)

🎮 ลองดู — pipeline ทำงานยังไง

พิมพ์ดู จะเห็น tier 1 (regex) ลองก่อน ถ้า fail ค่อยขึ้น tier 2 (LLM mock) ผลลัพธ์รวมท้ายสุด

เคล็ดลับลดต้นทุน

  • Cache อย่างน้อย 1024 tokens — system prompt ของเรามี few-shot examples ทำให้ผ่าน threshold
  • Cache TTL 5 นาที เพียงพอ ถ้าร้านมีคนพิมพ์อยู่บ่อย ๆ — cache hit rate > 80%
  • ตอบสั้น — output JSON ของเรา ~50 tokens เท่านั้น
  • Haiku 4.5 พอ — งาน parser ไม่ต้องใช้ Sonnet/Opus ยกเว้น vision
  • Batch API — สำหรับ workflow ที่ไม่ realtime (เช่น re-parse logs เก่า) ลดราคาอีก 50%

⚠️ ข้อควรระวังเมื่อใช้ LLM

  • ยืนยันก่อนเขียน DB — ถ้า confidence < 0.8 ให้บอทถาม "หมายถึงแบบนี้ใช่ไหม?" + ปุ่ม [ใช่] [ไม่ใช่]
  • Map item กับของในระบบ — Claude อาจสร้างชื่อใหม่ "พริกชี้ฟ้า" ทั้งที่เราเก็บไว้แค่ "พริก" → เทียบกับ items table ก่อน insert
  • Rate limit + retry — Claude API มี rate limit; ใช้ exponential backoff; ถ้า fail ให้ fallback ไป tier 4 (polite reply)
  • ห้ามเชื่อ qty ตรง ๆ ถ้าใหญ่ผิดปกติ — ถ้า qty > 10x ค่าเฉลี่ยปกติ → บอทถามยืนยันก่อน
  • Log ทุก LLM call — เก็บ input/output ลงตาราง llm_logs เพื่อตรวจ คาลิเบรต prompt ภายหลัง
  • Cost ceiling — ตั้ง daily cap ต่อ workspace เผื่อมีคนสแปม ป้องกัน bill ช็อก
  • ไม่ใส่ secret ใน prompt — system prompt มีแต่ schema ไม่มี customer data
เริ่มต้นด้วย Tier 1 (regex) ให้ใช้งานได้ก่อน — เปิด Tier 2 (LLM) ทีหลังเมื่อเริ่มเห็นข้อความที่ regex ทำไม่ได้ในชีวิตจริง
